import { Tab, Tabs } from "fumadocs-ui/components/tabs"
import { DynamicCodeBlock } from "fumadocs-ui/components/dynamic-codeblock"
import { latestVersion } from "@/lib/api.ts"
export const version = await latestVersion("holograms")

## Add our Repository

To start, add our repository to your project:

<Tabs defaultIndex='1' groupId="language" items={['Maven', 'Gradle Kotlin', 'Gradle Groovy', 'SBT']}>
  <Tab id="maven" value="Maven">
    ```xml
    <repository>
      <id>thenextlvl-releases</id>
      <name>TheNextLvl</name>
      <url>https://repo.thenextlvl.net/releases</url>
    </repository>
    ```
  </Tab>
  <Tab id="kotlin" value="Gradle Kotlin">
    ```kts
    maven {
      name = "thenextlvlReleases"
      url = uri("https://repo.thenextlvl.net/releases")
    }
    ```
  </Tab>
  <Tab id="groovy" value="Gradle Groovy">
    ```groovy
    maven {
      name "thenextlvlReleases"
      url "https://repo.thenextlvl.net/releases"
    }
    ```
  </Tab>
  <Tab id="sbt" value="SBT">
    ```scala
    resolvers +=
      "thenextlvl-releases"
        at "https://repo.thenextlvl.net/releases"
    ```
  </Tab>
</Tabs>

## Add the Holograms API

Add Holograms' API codebase as a dependency:

<Tabs
  defaultIndex="1"
  groupId="language"
  items={["Maven", "Gradle Kotlin", "Gradle Groovy", "SBT"]}
>
  <Tab id="maven" value="Maven">
    <DynamicCodeBlock
      lang="xml"
      code={`<dependency>
    <groupId>net.thenextlvl</groupId>
    <artifactId>holograms</artifactId>
    <version>${version}</version>
</dependency>`}
    />
  </Tab>
  <Tab id="kotlin" value="Gradle Kotlin">
    <DynamicCodeBlock lang="kts" code={`implementation("net.thenextlvl:holograms:${version}")`} />
  </Tab>
  <Tab id="groovy" value="Gradle Groovy">
    <DynamicCodeBlock
      lang="groovy"
      code={`implementation "net.thenextlvl:holograms:${version}"`}
    />
  </Tab>
  <Tab id="sbt" value="SBT">
    <DynamicCodeBlock lang="scala" code={`"net.thenextlvl" %% "holograms" %% "${version}"`} />
  </Tab>
</Tabs>

<Callout type="info">
  The version of the Holograms API is the same as the version of the Holograms plugin.  
  You can find all versions on the [repository](https://repo.thenextlvl.net/#/releases/net/thenextlvl/holograms).
</Callout>
